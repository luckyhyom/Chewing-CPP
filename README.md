CMake
```bash
wget https://github.com/Kitware/CMake/releases/download/v3.31.3/cmake-3.31.3.tar.gz
gunzip cmake-3.31.3.tar.gz 
tar -xvf cmake-3.31.3.tar 
cd cmake-3.31.3/
./bootstrap && make && sudo make install
```
